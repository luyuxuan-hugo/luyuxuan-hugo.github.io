快速排序

先选定一个x,将所有比x小的数放在左半部分,将所有比x大的数放在右半部分
再递归处理即可 

    #include <bits/stdc++>
    const int N = 1e6 + 10;//数据范围1~1000000
    int n;//数字个数
    int q[N];
    using namespace std;
    void quick_sort(int q[], int l, int r){
        //l是左边界,r是右边界
        if(l >= r) return;//如果范围内数组只有一个数或没有数,直接返回

        int x=q[(l+r)/2];//分界值x取数组中间值

        int i=l-1,j=r+1;

        while(i<j){

            while(q[++i]<x);//左指针右移 直到当前值大于x

            while(q[--j]>x);//右指针左移 直到当前值小于x

            if(i<j) swap(q[i],q[j]); //交换ij处的值
        }
        quick_sort(q,l,j);//递归处理左半部分

        quick_sort(q,j+1,r);//递归处理右半部分
    }
    int main(){
        scanf("%d", &n);

        for(int i=0;i<n;i++) scanf("%d", &q[i]);

        quick_sort(q,0,n-1);

        for(int i=0;i<n;i++) printf("%d, q[i]);

        return 0;
    }

如果分界值选择起点值q[l],数据过强会报错,故选择中间值q[(l+r)/2]

归并排序

将数组递归成只含一个数字的可排序的区间,最后在两个区间的数据从小到大放进tmp数组,再放进数组里

    #include <bits/stdc++.h>
    using namespace std;
    const int N = 1e5+10;
    int q[N],tmp[N];
    void merge_sort(int q[], int l,int r){
        if(l >= r) return;
        int mid = (l + r) >> 1;//取分界点
        merge_sort(q, l, mid),merge_sort(q, mid + 1, r);//递归到最小的区间,从最底层的两个数据排好序,再回到四个数据排好序,始终是两组数据之间的归并,  
                                                        //归并的规模从小到大,2->4->8...  

        int i = l,j=mid + 1,k = 0; //i是左起点,j是右起点  

        while(i <= mid && j <= r){

            if(q[i] <= q[j]) tmp[k++] = q[i++]; //两组数据在各自的内部都已经排好序,从小到大比较两组数据之间的大小就可以将整个数组排序  

            else tmp[k++] = q[j++];
        }//归并的过程,从小到大放入
        while(i <= mid) tmp[k++] = q[i++];

        while(j <= r) tmp[k++] = q[j++];//处理剩下的数据

        for(i = l,k = 0; i <= r; i++, k++) q[i] = tmp[k];//将临时数组中的数据取回  
    }
    int main(){
        int n;
        cin >> n;//n个数字
        for(int i = 0; i < n; i++) scanf("%d", &q[i]);//读入
        merge_sort(q , 0 , n-1);//排序
        for(int i = 0 ; i < n; i++) cout << q[i] << " ";//输出
    }

二分

在排好序的一组数据中,从中间的数据开始,如果中间的值比目标值小,则目标值在选取值的右边,反之则在左边 直到找到目标值

题目描述

给定一个按照升序排列的长度为 n 的整数数组，以及 q 个查询。

对于每个查询，返回一个元素 k 的起始位置和终止位置（位置从 0 开始计数）。

如果数组中不存在该元素，则返回 -1 -1。

输入格式

第一行包含整数 n 和 q，表示数组长度和询问个数。

第二行包含 n 个整数（均在 1∼10000 范围内），表示完整数组。

接下来 q 行，每行包含一个整数 k，表示一个询问元素。

输出格式

共 q 行，每行包含两个整数，表示所求元素的起始位置和终止位置。

如果数组中不存在该元素，则返回 -1 -1。

数据范围

1≤n≤100000
1≤q≤10000
1≤k≤10000

输入样例：

6 3
1 2 2 3 3 4
3
4
5

输出样例：

3 4
5 5
-1 -1 

    #include <bits/stdc++.h>
    const int N = 1e5+10;
    int n, m;//n个数字,询问个数m
    int q[N];
    using namespace std;
    int main(){
        scanf("%d%d",&n,&m);
    for(int i = 0; i < n; i++){
        scanf("%d",&q[i]);
        }
    while(m--){
        int x;//询问的数字x
        scanf("%d",&x);
        int l = 0,r = n - 1; //左右端点
    while(l<r){
            int mid = (l + r) >> 1;
            if(q[mid] >= x) r = mid;
            else l = mid + 1;
            
        }
        if(q[l]!=x) printf("-1 -1\n");
        else {
            printf("%d ",l);
            l = 0,r = n - 1;
            while(l<r){
                int mid = (l + r) + 1 >> 1 ; // 避免因为l + r 是奇数(即这组数据为偶数个)而引起的边界问题 
                                            //如果数据个数为偶数 每次取得的分界点应该不同 因此+1之后再/2(也可以>>1之后+0.5)
                if(q[mid] <= x) l = mid;
                else r = mid -1 ;
            }
            printf("%d\n", l);
        }
        
    }

    return 0;
        }

这道题是找目标数字的左右范围,即找到左起点和右终点

           else l = mid + 1;  
       }
